#include<stdio.h>
#include<stdlib.h>
typedef struct person {
    int num;
    struct person *next
}link;

link * ap(link *head); //按顺序接入链表
void display(link *head);//输出链表
void del(link *head);//删除所有节点
link * de(link *head,int n);//删除链表中指定节点
link * add(link *head, int n);//在链表中添加一个节点，假设链表已经按升序排好了

int main()
{
    link *head =NULL;
    for(int i=0;i<4;i++)
    {
       head = ap(head);    
    }
    display(head);
   
    int n;
    printf("\n你要删除的数字\n");
    scanf("%d",&n);
   
    head = de(head,n);
    display(head);
   
    printf("\n你要添加的数字\n");
    scanf("%d",&n);
   
    head = add(head,n);
    display(head);
   
    del(head);
    return 0;
}

link *ap(link *head)
{
    link *p=NULL,*pr =head;
    p = (link *) malloc(sizeof(link));
   
    if(p==NULL){
    printf("分配内存失败！");
    exit(0);
    }
   
    if (head == NULL)
    {
        head = p;
    }else{
        while(pr->next != NULL)
        {
            pr = pr->next;
        }
        pr->next = p;
    }
   
    int num;
    scanf("%d",&num);
   
    p->num = num;
    p->next = NULL;
   
    return head;
}

void display(link *head)
{
    link *p = head;
    while(p != NULL)
    {
        printf("%d ",p->num);
        p = p->next;
    }
}

void del(link *head)
{
    link *pr = NULL,*p =head;
    while(p != NULL)
    {
        pr = p;
        p = p-> next;
        free(pr);
    }
    //这里没有问题
}

link* de(link *head,int n)
{
    link *p = head,*pr = NULL;
   
    if(head==NULL)
    {
    printf("这是一个空链表\n");
    return head;
    }
   
    while(p->num != n && p->next != NULL)
    {
        pr = p;
        p = p->next;
    }
   
    if(p->num==n)
    {
            if (p == head)
            {
                    head = p-> next;
            }
            else{
                    pr->next = p ->next; 
                 }     
            free(p);
    }
    else{
    printf("没有查找到该数!\n");
    }
    return head;
}

link * add(link * head,int n)
{
    link *pr = head,*p = NULL;
    link *temp =NULL;
    p = (link *)malloc(sizeof(link));

	if (p == NULL)
	{
		printf ("分配内存失败！\n")
		return head;
	}
    p ->num = n;
    p ->next = NULL;
   
    while(pr->num < n && pr -> next != NULL)
    {
        temp = pr;
        pr = pr -> next;
    }
   
    if(pr == head)
    {
        p->next = pr;
        head = p;     
        return head;
        //在头上
    }
    if(pr->num >= n)
    {
    //在中间
        temp -> next = p;
        p->next  = pr;
       
    }else{
    //在尾巴上了
    pr->next = p;
    }
    return head;
}

package main

import "fmt"

type node struct {
	n    int
	next *node
}

//apone 在末尾加一个节点
func (head *node) apone() *node {
	p := new(node)
	pr := head
	if p == nil {
		fmt.Println("分配内存错误")
		return head
	}

	if head == nil {
		head = p
	} else {
		for pr.next != nil {
			pr = pr.next
		}
		pr.next = p
	}

	// 修改添加的节点的数据
	var n int
	fmt.Printf("请输入数字")
	fmt.Scan(&n)
	p.n = n

	return head
}

func (head *node) display() {
	p := head
	for p != nil {
		fmt.Printf("%d ", p.n)
		p = p.next
	}
	fmt.Println("输出完毕")
}

func (head *node) delone(n int) *node {
	var pr *node
	p := head

	for p.n != n && p.next != nil {
		pr = p
		p = p.next
	}

	if p.n == n {
		if p == head {
			head = p.next
		} else {
			pr.next = p.next
		}
	} else {
		fmt.Println("没有查找到该数")
	}

	return head
}

func (head *node) delall() {
	p := head
	pr := p

	for p != nil {
		pr = p
		p = p.next
		pr.next = nil
	}
	//手动释放头节点，虽然p没有指向这个节点但是head还指向它的
	//GC回收标准是内存没有指针指向
    //但是head为局部变量指针，所以这里的head是一个副本，在main函数中还是有一个不为 nil 的 head
	head = nil//无用代码
}

//addone 假设节点是按升序排号了的，我们在按升序插入一个节点
func (head *node) addone(n int) *node {
	var p, pr, temp *node
	temp = new(node)
	if temp == nil {
		fmt.Println("分配内存失败")
		return head
	}
	temp.n = n
	p = head

	if head == nil {
		return head
	}

	for p.n < n && p.next != nil {
		pr = p
		p = p.next
	}

	if p == head {
		temp.next = head
		head = temp
		return head
	}

	if p.n >= n {
		pr.next = temp
		temp.next = p
	} else {
		p.next = temp
	}

	return head
}

func main() {

	var head *node

	for i := 0; i < 7; i++ {
		head = head.apone()
	}
	head.display()

	var n int
	fmt.Println("输入一个你想删除的数字")
	fmt.Scan(&n)
	head = head.delone(n)
	head.display()

	fmt.Println("输入一个你想添加的数字")
	fmt.Scan(&n)
	head = head.addone(n)
	head.display()

	head.delall()

	head.display()

}
