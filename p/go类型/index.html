type

type IZ int

类型拷贝

type IZ = int

类型转换

typeB(typeA)

make和new的区别

new

new返回一个指向类型为 T，值为 0 的地址的指针

var p *[]int = new([]int)

// *p == nil

// with len and cap 0

p := new([]int)

new 和 &type 的不同处

new初始化的值为默认值，不支持自定义的值

&type初始化可以是自定义的值

new和&type的相同之处

返回的都是指针

make

make 返回一个类型为 T 的初始值

var v []int = make([]int, 10, 50)

v := make([]int, 10, 50)

make: 用于slice，map，和channel的初始化

整形

go语言中的int的大小是和操作系统位数相关的，如果是32位操作系统，int类型的大小就是4字节。

如果是64位操作系统，int类型的大小就是8个字节

一般情况

普通声明初始化

var name type

赋值

name = value

自定义声明初始化

var name type = value

编译器推断声明初始化

var name = value

简短声明初始化，实用于局部变量

name := value

字符类型

byte是uint 8的别名

自定义声明初始化

var ch byte = ‘A’

rune 是int32的别名

复数

complex 64 其中虚部实部为 float 32

complex128 其中虚部实部为 float 64

普通声明初始化

var name complex128

赋值

name = complex (x, y)

name = 888 + 999 i

自定义声明初始化

var name complex128 = complex(x, y)

简短声明初始化

name := complex(x, y)

name := 888 +666i

获取实部 real(name)

过去虚部 imag(name)

指针变量

一般声明初始化

var intpr *int

赋值

intpr = &n

自定义声明初始化

var intpr *int = &n

简短声明初始化

p := &n

数组

一般声明初始化

var arr [len]type

赋值

arr[i] = value

自定义声明

var arr = [5]int{1 8, 20, 15, 22, 16}

编译器推断声明初始化

var arr = […]int{ 5, 6, 7, 8, 22}

部分自定义声明初始化

var arr = [5]int{3: 222, 4: 999}

简短声明初始化

a := […]int{222，444，555，999}

切片

声明

var slice1 []type

赋值

slice1 []type = arr1[start:end]

slice1 = &arr1

slice1 = slicex[a :b]

// 不包含b但是包含a，切片元素个数为b-a

一般声明初始化

var slice1 []type = arr1[start:end]

slice1 := make([]type, start_length, capacity)

追加

slice1 = append(slice1, slice2…)

字典

声明

var map1 map[keytype]valuetype

初始化

map1 = map[string]int{“one”: 1, “two”: 2}

map1 = map[string]int{}

//初始化一个空字典

自定义声明初始化

var map1 map[string]int = map[string]int{“hunter”:12,“tony”:10}

推断声明初始化

var map1 = map[string]int{“one”: 1, “two”: 2}

简短声明初始化

map1 = map[string]int{“one”: 1, “two”: 2}

make声明初始化

mapHaiCoder := make(map[string]string, 3)

//3是容量

超出容量时会自动扩容，但尽量提供一个合理的初始值

赋值

map1[key1] = val1

读取

v := map1[key1]

删除

delete(map1,kay)

结构体

定义一个自己的结构体

type T struct {

field1 type1

field2 type2

…

}

声明自己的结构体

var s T

结构体初始化

s = struct1{10, 15.5, “Chris”}

修改结构体内部字段

s.a = 5

s.b = 8

结构体指针

&struct1{a, b, c} 是一种简写，底层仍然会调用 new ()

结构体内嵌结构体

type innerS struct {

in1 int

in2 int

}

type outerS struct {

b int

c float32

// anonymous field

int

//anonymous field

innerS

}

func main() {

outer := new(outerS)

outer.b = 6

outer.c = 7.5

outer.in1 = 5

outer.in2 = 1

}

结构体命名冲突

type A struct {a int}

type B struct {a, b int}

type C struct {A; B}

var c C

报错 c.a，编译器不知道是A中的还是B中的

正确 c.A.a

接口

接口是一种类型，它规定了变量有哪些方法

定义一个接口类型

type name interface{

FuncName(arg)return

xxx

}

声明某个接口类型

var T name

接口赋值

​ x = a

T = new(NokiaPhone)

GO方法有自动解引，是传值还是引用以方法为准

使用接口

T.FuncName

任意类型都被认为实现了空接口

接口方法最好不要超过三个

每个接口变量在内存中占据两个字长：

一个用来存储它包含的类型，另一个用来存储它包含的数据或者指向数据的指针

信道

声明一个信道(双向)

var a chan int

信道初始化

a = make(chan int)

简洁声明初始化

a := make(chan int)

有10个缓冲的信道

a := make(chan int, 10)

// 读取信道 a

data := <- a

// 写入信道 a

a <- data

单向信道

只读

var pipline = make(chan int)

type Receiver = <-chan int

var receiver Receiver = pipline

只写

var pipline = make(chan int)

type Sender = chan<- int

var sender Sender = pipline

关闭

close(pipline)
